Paper examples

using Convex, ECOS

Summation.

println("Summation example")
x = Variable();
e = 0;
@time begin
  for i = 1:1000
    global e
    e += x;
  end
  p = minimize(e, x>=1);
end
@time solve!(p, () -> ECOS.Optimizer(verbose=0))
Summation example
  0.017194 seconds (4.64 k allocations: 11.763 MiB)
  1.627143 seconds (2.10 M allocations: 104.293 MiB, 7.55% gc time)

Indexing.

println("Indexing example")
x = Variable(1000, 1);
e = 0;
@time begin
  for i = 1:1000
    global e
    e += x[i];
  end
  p = minimize(e, x >= ones(1000, 1));
end
@time solve!(p, () -> ECOS.Optimizer(verbose=0))
Indexing example
  0.194817 seconds (125.12 k allocations: 17.627 MiB, 14.98% gc time)
  0.416668 seconds (586.92 k allocations: 101.890 MiB, 21.10% gc time)

Matrix constraints.

println("Matrix constraint example")
n, m, p = 100, 100, 100
X = Variable(m, n);
A = randn(p, m);
b = randn(p, n);
@time begin
  p = minimize(norm(vec(X)), A * X == b);
end
@time solve!(p, () -> ECOS.Optimizer(verbose=0))
Matrix constraint example
  0.245197 seconds (259.30 k allocations: 12.123 MiB, 21.24% gc time)
  3.571076 seconds (851.04 k allocations: 340.265 MiB, 29.73% gc time)

Transpose.

println("Transpose example")
X = Variable(5, 5);
A = randn(5, 5);
@time begin
  p = minimize(norm2(X - A), X' == X);
end
@time solve!(p, () -> ECOS.Optimizer(verbose=0))
