Paper examples

using Convex, ECOS

Summation.

println("Summation example")
x = Variable();
e = 0;
@time begin
  for i = 1:1000
    global e
    e += x;
  end
  p = minimize(e, x>=1);
end
@time solve!(p, ECOS.Optimizer(verbose=0))
Summation example
  0.029519 seconds (4.64 k allocations: 11.764 MiB, 51.66% gc time)
  1.422886 seconds (1.97 M allocations: 98.322 MiB, 5.61% gc time)

Indexing.

println("Indexing example")
x = Variable(1000, 1);
e = 0;
@time begin
  for i = 1:1000
    global e
    e += x[i];
  end
  p = minimize(e, x >= ones(1000, 1));
end
@time solve!(p, ECOS.Optimizer(verbose=0))

  0.177569 seconds (125.79 k allocations: 17.665 MiB, 10.41% gc time)
  0.127379 seconds (169.34 k allocations: 79.757 MiB, 42.65% gc time)

Matrix constraints.

println("Matrix constraint example")
n, m, p = 100, 100, 100
X = Variable(m, n);
A = randn(p, m);
b = randn(p, n);
@time begin
  p = minimize(norm(vec(X)), A * X == b);
end
@time solve!(p, ECOS.Optimizer(verbose=0))
Matrix constraint example
  0.224769 seconds (262.37 k allocations: 12.286 MiB, 20.48% gc time)
  3.635303 seconds (3.05 M allocations: 498.453 MiB, 28.71% gc time)

Transpose.

println("Transpose example")
X = Variable(5, 5);
A = randn(5, 5);
@time begin
  p = minimize(norm2(X - A), X' == X);
